嵌入式代码题只背这个

五个结构体速记

不要硬背整段代码。考试写程序题时，先把结构体字段写对，再按“开时钟、填结构体、初始化、使能”的顺序套模板。

总口诀：GPIO 是“针速模”，TIM 是“周分分模”，PWM 是“模使脉极”，NVIC 是“道抢响使”，EXTI 是“线式触使”。

五个结构体

1

GPIO_InitTypeDef

口诀：GPIO 针速模

看到 GPIO 输出、输入、LED、按键，先写这个。GPIO 必会。

  • GPIO_Pin 选引脚
  • GPIO_Speed 选速度
  • GPIO_Mode 选模式
GPIO_InitTypeDef GPIO_InitStructure;

RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOC, ENABLE);

GPIO_InitStructure.GPIO_Pin = GPIO_Pin_1;
GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
GPIO_Init(GPIOC, &GPIO_InitStructure);
2

TIM_TimeBaseInitTypeDef

口诀：时基周分分模

看到定时、延时、溢出时间，写这个。公式先行：T = (PSC+1)(ARR+1)/TIMx_CLK。

  • TIM_Period 周期 ARR
  • TIM_Prescaler 预分频 PSC
  • TIM_ClockDivision 时钟分割
  • TIM_CounterMode 计数模式
TIM_TimeBaseInitTypeDef TIM_TimeBaseStructure;

RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM2, ENABLE);

TIM_TimeBaseStructure.TIM_Period = 720 - 1;
TIM_TimeBaseStructure.TIM_Prescaler = 100 - 1;
TIM_TimeBaseStructure.TIM_ClockDivision = TIM_CKD_DIV1;
TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
TIM_TimeBaseInit(TIM2, &TIM_TimeBaseStructure);

TIM_Cmd(TIM2, ENABLE);
3

TIM_OCInitTypeDef

口诀：OC 模使脉极

看到 PWM、输出比较、呼吸灯、风扇调速，写这个。Pulse 就是 CCR，占空比看它。

  • TIM_OCMode PWM/比较模式
  • TIM_OutputState 输出使能
  • TIM_Pulse 脉冲值 CCR
  • TIM_OCPolarity 输出极性
TIM_OCInitTypeDef TIM_OCInitStructure;

TIM_OCInitStructure.TIM_OCMode = TIM_OCMode_PWM1;
TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Enable;
TIM_OCInitStructure.TIM_Pulse = 500;
TIM_OCInitStructure.TIM_OCPolarity = TIM_OCPolarity_High;
TIM_OC1Init(TIM3, &TIM_OCInitStructure);

TIM_OC1PreloadConfig(TIM3, TIM_OCPreload_Enable);
4

NVIC_InitTypeDef

口诀：NVIC 道抢响使

看到中断优先级，写这个。嵌套看抢占优先级，响应优先级只管同级先后。

  • NVIC_IRQChannel 中断通道
  • NVIC_IRQChannelPreemptionPriority 抢占优先级
  • NVIC_IRQChannelSubPriority 响应优先级
  • NVIC_IRQChannelCmd 通道使能
NVIC_InitTypeDef NVIC_InitStructure;

NVIC_InitStructure.NVIC_IRQChannel = EXTI0_IRQn;
NVIC_InitStructure.NVIC_IRQChannelPreemptionPriority = 1;
NVIC_InitStructure.NVIC_IRQChannelSubPriority = 1;
NVIC_InitStructure.NVIC_IRQChannelCmd = ENABLE;
NVIC_Init(&NVIC_InitStructure);
5

EXTI_InitTypeDef

口诀：EXTI 线式触使

看到外部中断、按键中断，写这个。先 GPIO_EXTILineConfig，再 EXTI_Init。

  • EXTI_Line 中断线
  • EXTI_Mode 中断/事件模式
  • EXTI_Trigger 触发边沿
  • EXTI_LineCmd 线路使能
EXTI_InitTypeDef EXTI_InitStructure;

GPIO_EXTILineConfig(GPIO_PortSourceGPIOA, GPIO_PinSource0);

EXTI_InitStructure.EXTI_Line = EXTI_Line0;
EXTI_InitStructure.EXTI_Mode = EXTI_Mode_Interrupt;
EXTI_InitStructure.EXTI_Trigger = EXTI_Trigger_Falling;
EXTI_InitStructure.EXTI_LineCmd = ENABLE;
EXTI_Init(&EXTI_InitStructure);

其他常考代码模板

PC.1 小灯闪烁

口诀：开 C 配 PC1，Reset 亮，Set 灭，Delay 夹中间。

void Delay(vu32 nCount)
{
    for (; nCount != 0; nCount--);
}

int main(void)
{
    GPIO_InitTypeDef GPIO_InitStructure;

    RCC_APB2PeriphClockCmd(RCC_APB2Periph_GPIOC, ENABLE);
    GPIO_InitStructure.GPIO_Pin = GPIO_Pin_1;
    GPIO_InitStructure.GPIO_Mode = GPIO_Mode_Out_PP;
    GPIO_InitStructure.GPIO_Speed = GPIO_Speed_50MHz;
    GPIO_Init(GPIOC, &GPIO_InitStructure);

    while (1) {
        GPIO_ResetBits(GPIOC, GPIO_Pin_1);
        Delay(0x1FFFFF);
        GPIO_SetBits(GPIOC, GPIO_Pin_1);
        Delay(0x1FFFFF);
    }
}

TIM2 1ms 精确定时

口诀：先算再配。T = 100 * 720 / 72MHz = 1ms。

void TIM2_Init(void)
{
    TIM_TimeBaseInitTypeDef TIM_TimeBaseStructure;

    RCC_APB1PeriphClockCmd(RCC_APB1Periph_TIM2, ENABLE);
    TIM_TimeBaseStructure.TIM_Period = 720 - 1;
    TIM_TimeBaseStructure.TIM_Prescaler = 100 - 1;
    TIM_TimeBaseStructure.TIM_ClockDivision = TIM_CKD_DIV1;
    TIM_TimeBaseStructure.TIM_CounterMode = TIM_CounterMode_Up;
    TIM_TimeBaseInit(TIM2, &TIM_TimeBaseStructure);

    TIM_ClearFlag(TIM2, TIM_FLAG_Update);
    TIM_Cmd(TIM2, ENABLE);
}

中断服务函数固定写法

口诀：判断标志，处理任务，清除标志。

void EXTI0_IRQHandler(void)
{
    if (EXTI_GetITStatus(EXTI_Line0) != RESET) {
        /* 在这里写中断处理代码 */
        EXTI_ClearITPendingBit(EXTI_Line0);
    }
}

PWM 输出最小骨架

口诀：GPIO 复用推挽，TIM 定周期，OC 写占空比。

GPIO_InitStructure.GPIO_Mode = GPIO_Mode_AF_PP;

TIM_TimeBaseStructure.TIM_Period = 1000 - 1;
TIM_TimeBaseStructure.TIM_Prescaler = 72 - 1;
TIM_TimeBaseInit(TIM3, &TIM_TimeBaseStructure);

TIM_OCInitStructure.TIM_OCMode = TIM_OCMode_PWM1;
TIM_OCInitStructure.TIM_OutputState = TIM_OutputState_Enable;
TIM_OCInitStructure.TIM_Pulse = 500;
TIM_OCInitStructure.TIM_OCPolarity = TIM_OCPolarity_High;
TIM_OC1Init(TIM3, &TIM_OCInitStructure);
代码题答题顺序：先写结构体变量，再开外设时钟，然后填字段，调用 Init，最后清标志、开中断或使能外设。背顺序，比背整段代码更稳。
